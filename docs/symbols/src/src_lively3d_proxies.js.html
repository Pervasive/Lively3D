 45 			Serializes Desktop state as JSON and uploads it to the dropbox.

 47 		*/
 48 		SaveDesktop: function(filename){
 49 		
 50 			var LivelyState = '';
 51 			var state = [];
 52 			var Applications = Lively3D.GetApplications();
 53 			for ( var i in Applications ){
 54 				if ( Applications.hasOwnProperty(i)){
 55 					var app = Applications[i];
 56 					var maximized = app.isMaximized();
 57 					if ( app.isMaximized() == true ){
 58 						Lively3D.Minimize(app);
 59 					}
 60 					var AppJSON = {
 61 						Name: app.GetName(),
 62 						Location: { x: app.GetCurrentSceneObject().getLocX(), y: app.GetCurrentSceneObject().getLocY(), z: app.GetCurrentSceneObject().getLocZ()},
 63 						Rotation: app.GetCurrentSceneObject().getRotation(),
 64 						Closed: app.isClosed(),
 65 						Maximized: maximized,
 66 						Code: app.GetApplicationCode().toString(),
 67 						Init: app.GetInitializationCode().toString(),
 68 						AppState: app.Save()
 69 					}
 70 					
 71 					state.push(AppJSON);
 72 					if ( maximized == true ){
 73 						Lively3D.Maximize(app);
 74 					}
 75 				}
 76 			}
 77 			LivelyState = JSON.stringify(state);
 78 			Lively3D.FileOperations.uploadScript(filename, LivelyState, "states/" + Lively3D.GetUsername());
 79 		},

138 	var ParseDesktopJSON = function(data){
139 		var JSONArray = JSON.parse(data);
140 		
141 		for ( var i in JSONArray ){
142 			if ( JSONArray.hasOwnProperty(i)){
143 				var JSONObject = JSONArray[i];
144 				var CodeFunc = eval("(" + JSONObject.Code + ")");
145 				var InitFunc = eval("(" + JSONObject.Init + ")");
146 				var app = Lively3D.AddApplication(JSONObject.Name, CodeFunc, InitFunc);
147 				SetAppLocation(app, JSONObject.Location);
148 				SetAppRotation(app, JSONObject.Rotation);
149 				app.StateFromDropbox = true;
150 				
151 				
152 				if ( JSONObject.Closed != true ){
153 					//Lively3D.Open(app);
154 					app.OpenAfterLoad = true;
155 				}
156 				
157 				if ( JSONObject.Maximized == true ){
158 					//Lively3D.Maximize(app);
159 					app.MaximizeAfterLoad = true;
160 				}
161 				
162 				app.Load(JSONObject.AppState);
163 				app.StartApp();
164 			}
165 		}
166 	};

429 		CheckAvailability: function(callback){
430 		
431 			$.ajax({
432 				url: "http://localhost:8000/", 
433 				success: function(data){
434 					if ( data.success == false ){
435 						Lively3D.UI.ShowMessage(data.message);
436 					}else{
437 						callback()
438 					}
439 				},
440 				error: function(){

467 		*/

469 			$.get("http://localhost:8000/filearray", {'names[]': App.Resources[resource], path: App.ResourcePath}, function(array){
