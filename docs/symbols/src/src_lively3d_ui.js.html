211 	Lively3D.UI.ActiveDialog;
212 	/**
213 		Shows HTML content to user.
214 		@param content HTML code to display.
215 		@param {boolean} omitCancel Whether to display cancel-button or not. If true, button is omitted.
216 	*/
217 	Lively3D.UI.ShowHTML = function(content, omitCancel){
218 		if ( this.ActiveDialog != null ){
219 			if ( this.ActiveDialog.cancelOmitted != true){
220 				SwitchDialog(content);
221 			}
222 		}
223 		else{
224 			if ( omitCancel != true ){
225 				var dialog = $("<div class='dialog' id='dialog'><h3 onclick='Lively3D.UI.CloseDialog();'>Cancel</h3></div>");
226 			}
227 			else{
228 				var dialog = $("<div class='dialog' id='dialog'></div>");
229 				dialog.cancelOmitted = true;
230 			}
231 			dialog.prepend(content);
232 			dialog.appendTo($('#container'));
233 			dialog.fadeIn("slow");
234 			this.ActiveDialog = dialog;
235 		}
236 	}
237 	
238 	/**
239 		Closes dialog.
240 	*/
241 	Lively3D.UI.CloseDialog = function(){
242 		this.ActiveDialog.fadeOut("slow", function(){
243 			Lively3D.UI.ActiveDialog.remove();
244 			Lively3D.UI.ActiveDialog = null;
245 		});
246 	}
247 	
248 	/**
249 		Switches existing dialog.
250 		@param content Contents of new dialog.
251 	*/
252 	var SwitchDialog = function(content){
253 		Lively3D.UI.ActiveDialog.fadeOut("fast", function(){
254 			Lively3D.UI.ActiveDialog.remove();
255 			var dialog = $("<div class='dialog' id='dialog'><h3 onclick='Lively3D.UI.CloseDialog();'>Cancel</h3></div>");
256 			dialog.prepend(content);
257 			dialog.appendTo($('#container'));
258 			dialog.fadeIn("fast");
259 			Lively3D.UI.ActiveDialog = dialog;
260 		});
261 	}
