364 		var ray = scene.makeRay(window.innerWidth/2, window.innerHeight/2);

366 		FrontOfCamera = [ray.origin[0] - ray.coord[0]*11,

368 			ray.origin[2] - ray.coord[2]*11];
369 			
370 		app.OldLocation = [app.GetWindowObject().getLocX(),app.GetWindowObject().getLocY(),app.GetWindowObject().getLocZ()];
371 		app.GetWindowObject().setLoc(FrontOfCamera[0],FrontOfCamera[1],FrontOfCamera[2]).setScale(1,1,1);

405 	/**
406 		Fires given event to given application.
407 		@param {string} eventname Name of the event for firing.
408 		@param app Application which is the target of the event.
409 		@param textureCoordinates x- and y-coordinates in texture of the event.
410 		@param glge_object The GLGE Object which represents the target application.
411 		@param event Javascript event-object.
412 	*/
413 	Lively3D.FireEvent= function(eventname, app, textureCoordinates,glge_object, event ){
414 	
415 		
416 		if ( textureCoordinates, glge_object ){
417 			var material;
418 			material = glge_object.getMaterial();
419 			
420 			var tex = material.getLayers()[0].getTexture();
421 			
422 			//tex is GLGE.TextureCanvas
423 			if ( tex.getCanvas ){
424 				var params;
425 				if ( eventname != "mousewheel" ){
426 					var canvas = tex.getCanvas();
427 					var canvasCoordinates = [ textureCoordinates[0]*canvas.width, (1-textureCoordinates[1])*canvas.height ]; 
428 					params = { "coord":canvasCoordinates, "canvas":canvas };
429 				}
430 				else{
431 					params = event;
432 				}
433 				material.fireEvent(eventname, params);
434 			}
435 		}
436 		else{
437 			app.GetCurrentSceneObject().fireEvent(eventname);
438 		}
439 	}

442 		Picks mouse location within 3D scene. 
443 		@param that this context
444 		@param mouseevent JavasScript mouse event.
445 		@returns Location object which includes GLGE object of the coordinates, distance of the object, texture coordinates etc.
446 	*/
447 	Lively3D.PickLocation = function(that, mouseevent){
448 		var pos = findPos(that);
449 		var x = mouseevent.pageX - pos.x;
450 		var y = mouseevent.pageY - pos.y;
451 		
452 		var loc = Scenes[CurrentScene].GetScene().pick(x, y)
453 		
454 		return loc;
455 	}

457 	var findPos = function(obj) {
458 		var curleft = curtop = 0;
459 		if (obj.offsetParent) {
460 			do {
461 				curleft += obj.offsetLeft;
462 				curtop += obj.offsetTop;
463 			} while (obj = obj.offsetParent);
464 			return { x: curleft, y: curtop };
465 		}
466 	}

561 	var DefaultCanvasEvents = function(canvas){
562 		canvas.onclick = function(e){
563 			
564 			var loc = Lively3D.PickLocation(this,e);
565 			if (loc != null){
566 				var app = Lively3D.GetApplication(loc.object.parent);
567 
568 				if ( app != null ){
569 
570 					if ( app.CheckPermission(Lively3D.PERMISSIONS.CLOSE, loc.object)){
571 						Lively3D.Close(app);
572 					}
573 					
574 					if ( !app.isClosed() ){
575 						Lively3D.FireEvent('click', app, loc.texture, loc.object);
576 					}
577 				}
578 			}
579 		}
580 		
581 		canvas.ondblclick = function(e){
582 			var loc = Lively3D.PickLocation(this, e);
583 			console.log("dblclick with loc: " + loc);
584 			if ( loc != null ){
585 				var app = Lively3D.GetApplication(loc.object.parent);
586 					
587 				if ( app != null){
588 					
589 					if ( app.CheckPermission(Lively3D.PERMISSIONS.OPEN, loc.object) ){
590 						Lively3D.Open(app);
591 					}
592 					else if (app.CheckPermission(Lively3D.PERMISSIONS.CLOSE, loc.object) ){
593 						Lively3D.Close(app);
594 					}
595 					
596 					else if (app.CheckPermission(Lively3D.PERMISSIONS.MAXIMIZE, loc.object)){
597 						if ( !app.isMaximized() ){
598 							Lively3D.Maximize(app);
599 						}
600 						else{
601 							Lively3D.Minimize(app);
602 						}
603 					}
604 				}
605 			}
606 		}

608 		canvas.onmousedown = function(e){
609 			
610 			var loc = Lively3D.PickLocation(this,e);
611 			if ( loc != null ){
612 				var app = Lively3D.GetApplication(loc.object.parent);
613 				
614 				if ( app != null ){
615 					if ( e.which == 3 ){
616 						if ( !app.isClosed() ){
617 							Lively3D.FireEvent('rightclick', app, loc.texture, loc.object);
618 						}	
619 					}
620 					else{
621 						if ( app.CheckPermission(Lively3D.PERMISSIONS.DRAG, loc.object) && (app.isClosed() || !app.isMaximized())){
622 							Lively3D.GLGE.draggableObject = app.GetCurrentSceneObject();
623 						}
624 						if ( !app.isClosed() ){
625 							Lively3D.GLGE.clickedObject = app.GetWindowMaterial();
626 							Lively3D.FireEvent('mousedown', app, loc.texture, loc.object);
627 						}
628 					}
629 				}
630 			}
631 		}

725 	var SceneLoader = function(url){
726 		
727 		var parsedURL = parseUrl(url);
728 		
729 		console.log("scene loaded from url: "+ url + " and file is " + parsedURL.file);
730 		
731 		var found = false;
732 		for ( var i in SceneBuffer){
733 			if ( SceneBuffer.hasOwnProperty(i)){
734 				if ( SceneBuffer[i].File == parsedURL.file){
735 					if ( SceneBuffer[i].DocumentDone ){
736 						SceneBuffer[i].DocumentDone();
737 					}
738 					Scenes.push(new Lively3D.Scene().SetScene(Lively3D.GLGE.document.getElement(SceneBuffer[i].Id)).SetModel(SceneBuffer[i]));
739 					AddAppsToScene(SceneBuffer[i]);
740 					SceneBuffer.splice(i, 1);
741 					found = true;
742 				}
743 			}
744 		}
745 		if ( found == false){
746 			console.log("error loading scene");
747 		}

751 		var e=/((http|ftp):\/)?\/?([^:\/\s]+)((\/\w+)*\/)([\w\-\.]+\.[^#?\s]+)(#[\w\-]+)?/;
752 
753 		if (data.match(e)) {
754 			return  {url: RegExp['$&'],
755                 protocol: RegExp.$2,
756                 host:RegExp.$3,
757                 path:RegExp.$4,
758                 file:RegExp.$6,
759                 hash:RegExp.$7};
760 		}
761 		else {
762 			return  {url:"", protocol:"",host:"",path:"",file:"",hash:""};
763 		}
